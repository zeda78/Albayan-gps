from flask import Flask, request, jsonify, render_template_string
import math
import os

app = Flask(__name__)

# معادلة حساب المسافة بناءً على المعايرة الميدانية (0.92)
def get_dist(sig): 
    return (10**(((61 - float(sig)) - 55) / 44.9) * 1000) * 0.92

@app.route('/calc', methods=['POST'])
def calc():
    d = request.json
    lat, lon = float(d['lat']), float(d['lon'])
    dist = get_dist(-80)
    nlat = lat + (dist/111000)
    nlon = lon + (dist/(111000*math.cos(math.radians(lat))))
    return jsonify({'lat': round(nlat, 6), 'lon': round(nlon, 6)})
